3. Java Program to Create a Button and Display Image in the Frame when Clicked

import java.awt.*;
import java.awt.event.*;

class myFrame extends Frame {
Button b1 = new Button("Click to show image");
Image img;
myFrame() {
b1.addActionListener(new ActionListener() {
public void actionPerformed(ActionEvent e) {
img = Toolkit.getDefaultToolkit().getImage("path/to/image.jpg");
repaint();
}
});
add(b1);
setLayout(new FlowLayout());
setSize(500, 500);
setTitle("Display Image on Button Click");
setVisible(true);
}
public void paint(Graphics g) {
if (img != null) {
g.drawImage(img, 50, 50, this);
}
}
public static void main(String ap[]) {
myFrame f1 = new myFrame();
}
}

19. Java Program to Validate the TextField for only entering numbers using KeyListener

 import java.awt.*;
 import java.awt.event.*;
 class NumberValidator extends Frame {
   TextField tf = new TextField();
   NumberValidator(){
     add(tf);
     setSize(300,200);
     setVisible(true);
     tf.addKeyListener(new KeyAdapter(){
       public void keyTyped(KeyEvent ke){
         if(!Character.isDigit(ke.getKeyChar())){
           ke.consume();
         }
       }
       });
   }

     new NumberValidator();
   }
   }

20. Java Program to Validate the TextField for only entering email id using KeyAdapter

 import java.awt.*;
 import java.awt.event.*;
 import java.util.regex.*;
 class EmailValidator extends Frame {
   TextField tf = new TextField();
   EmailValidator(){
     add(tf);
     setSize(300,200);
     setVisible(true);
     tf.addKeyListener(new KeyAdapter(){
       public void keyTyped(KeyEvent ke){
         String text = tf.getText();
         if(!text.matches("[a-zA-Z0-9@.]*")){
           ke.consume();
         }
       }
       });
   }
   public static void main(String ap[]){
     new EmailValidator();
   }
   }
