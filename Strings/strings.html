16. Program to count the total number of vowels and consonants in a string

class Main{
public static void main(String[] args) {
String input = "Hello World";
int vowels = 0, consonants = 0;
char[] charArray = input.toCharArray();
int i;
for ( i = 0; i < charArray.length; i++) {
char c = charArray[i];
if ((c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z')) {
if ("AEIOUaeiou".indexOf(c) != -1) {
vowels++;
} else {
consonants++;
}
}
}
System.out.println("Vowels: " + vowels + ", Consonants: " + consonants);
}
}

*27. Find the Largest and Smallest Word in a String

class Main {
public static void main(String[] args) {
String input = "This is a normal program of java written by me";
String[] words = input.split("\\s+");
String largest = "", smallest = words[0];
int i;
for (i = 0; i < words.length; i++) {
String word = words[i];
if (word.length() > largest.length()) {
largest = word;
}
if (word.length() < smallest.length()) {
smallest = word;
}
}

System.out.println("Largest Word: " + largest);
System.out.println("Smallest Word: " + smallest);
}
}

46. Reverse String Word by Word

String input = "Hello World";
String[] words = input.split(" ");
String reversedWords = "";

for( i = words.length - 1; i >= 0; i--) {
reversedWords += words[i];
if (i > 0) {
reversedWords += " ";
}
}

System.out.println("Reversed String Word by Word: " + reversedWords);
}
}
